    const hw = (routingItem.width / 2) + padding;

    const hh = (routingItem.height / 2) + padding;

    const scaleX = Math.abs(dx) > 0.01 ? hw / Math.abs(dx) : Infinity;

    const scaleY = Math.abs(dy) > 0.01 ? hh / Math.abs(dy) : Infinity;

    const scale = Math.min(scaleX, scaleY);

    const finalScale = Math.min(scale, 0.85);

    return { x: cx + dx * finalScale, y: cy + dy * finalScale };

  };


  const getOrganicPath = (x1, y1, x2, y2, isCurved, hasArrow) => {

    let dx = x2 - x1; let dy = y2 - y1; let dist = Math.sqrt(dx * dx + dy * dy);

    if (dist < 10) return ''; 

    let path = ''; let angle = 0;

    if (!isCurved || dist < 40) {

      path = `M ${x1} ${y1} L ${x2} ${y2}`; angle = Math.atan2(dy, dx);

    } else {

      const nx = -dy / dist; const ny = dx / dist;

      const curveDepth = Math.min(60, dist * 0.18); 

      const cx = (x1 + x2) / 2 + nx * curveDepth;

      const cy = (y1 + y2) / 2 + ny * curveDepth;

      path = `M ${x1} ${y1} Q ${cx} ${cy} ${x2} ${y2}`;

      angle = Math.atan2(y2 - cy, x2 - cx);

    }

    if (hasArrow) {

      const headLength = 16; 

      const a1 = angle - Math.PI / 6.5; const a2 = angle + Math.PI / 5.5; 

      const h1x = x2 - headLength * Math.cos(a1); const h1y = y2 - headLength * Math.sin(a1);

      const h2x = x2 - headLength * Math.cos(a2); const h2y = y2 - headLength * Math.sin(a2);

      path += ` M ${x2} ${y2} Q ${x2 - headLength*0.4 * Math.cos(angle)} ${y2 - headLength*0.4 * Math.sin(angle)} ${h1x} ${h1y}`;

      path += ` M ${x2} ${y2} Q ${x2 - headLength*0.4 * Math.cos(angle)} ${y2 - headLength*0.4 * Math.sin(angle)} ${h2x} ${h2y}`;

    }

    return path;

  };
